import { excelToJson, jsonToExcel} from 't-comm/es/sheet/index';

excelToJson(params) ​

描述：

excel 转 json

参数：

参数名类型描述
paramsobject

参数

返回:

jsonData

示例

typescript
const options = {
  header: ['id', 'name', 'age'], // 可选：自定义表头
  range: 1,                      // 可选：跳过第一行（标题行）
  defval: null,                  // 可选：空单元格的默认值
  raw: false,                    // 可选：是否保留原始数据格式
};

excelToJson({
  filePath: CONFIG.xlsxPath,
  sheetIndex: 1,
  options,
});

// [
//   { id: 1, name: '2', age: '3' },
//   { id: 1, name: '2', age: '3' }
// ];

jsonToExcel(params) ​

描述：

json 转 excel

参数：

参数名类型描述
paramsobject

参数

示例

typescript
{ workbook, worksheet }

const jsonData = [
  { id: 1, name: '2', age: '3' },
  { id: 1, name: '2', age: '3' },
];

jsonToExcel({
  jsonData,
  outputPath: CONFIG.outputFilePath,
  options: {
    header: ['id', 'name', 'age'],  // 可选：自定义表头顺序
    skipHeader: false,              // 可选：是否跳过表头行
  },
  sheetName: 'addedData',
});
